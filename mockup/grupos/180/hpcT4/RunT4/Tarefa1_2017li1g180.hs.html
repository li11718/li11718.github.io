   41 processa :: Caminho -> Orientacao -> Altura -> (Int,Int) -> [[Peca]] -> [[Peca]]
   42 processa [] _ _ _ m = m
   43 processa (CurvaDir:c) d a (x,y) m = processa c d' a (mexe (x,y) d') m'
   44     where m' = replace m (x,y) (blocoCurvo d d' a)
   45           d' = roda d True
   46 processa (CurvaEsq:c) d a (x,y) m = processa c d' a (mexe (x,y) d') m'
   47     where m' = replace m (x,y) (blocoCurvo d d' a)
   48           d' = roda d False
   49 processa (Avanca:c) d a (x,y) m = processa c d a (mexe (x,y) d) m'
   50     where m' = replace m (x,y) (Peca Recta a)
   51 processa (s:c) d a (x,y) m = processa c d a' (mexe (x,y) d) m'
   52     where m' = replace m (x,y) p'
   53           a' = adapta s a
   54           p' = (blocoRampa s d) (min a a')

   56 replace :: [[a]] -> (Int,Int) -> a -> [[a]]
   57 replace m (x,y) e = (take y m) ++ [l] ++ (drop (y+1) m)
   58     where l = (take x (atNote "replace1" m y)) ++ [e] ++ (drop (x+1) (atNote "replace2" m y))

   60 blocoCurvo :: Orientacao -> Orientacao -> Altura -> Peca
   61 blocoCurvo Norte Este  = Peca (Curva Norte) 
   62 blocoCurvo Este Sul    = Peca (Curva Este)  
   63 blocoCurvo Sul Oeste   = Peca (Curva Sul)   
   64 blocoCurvo Oeste Norte = Peca (Curva Oeste) 
   65 blocoCurvo m n = blocoCurvo (roda (roda n True) True) (roda (roda m True) True)
   66 -- Este Norte == Sul Oeste

   69 adapta :: Passo -> Altura -> Altura
   70 adapta Sobe  a = a+1
   71 adapta Desce a = a-1

   74 blocoRampa :: Passo -> Orientacao -> (Altura -> Peca)
   75 blocoRampa Sobe Norte = Peca (Rampa Norte)
   76 blocoRampa Sobe Oeste = Peca (Rampa Oeste)
   77 blocoRampa Sobe Sul   = Peca (Rampa Sul)
   78 blocoRampa Sobe Este  = Peca (Rampa Este)
   79 blocoRampa Desce d    = blocoRampa Sobe (roda (roda d True) True)
